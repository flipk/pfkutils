# Each block: 16 bytes header + (350 * 4 bytes) data = 1416 bytes per record
splot 'data.bin' binary \
    endian=big \
    format='%*2int16%int16%int16' \
    record=350 \
    skip=16 \
    with lines

# Explanation of the Options:
#  record=350: This tells Gnuplot that each "record" (or block)
#              contains exactly 350 data points. Gnuplot will automatically
#              handle the headers if configured correctly.
#  skip=16:    This tells Gnuplot to skip 16 bytes at the start
#              of each record.
#  format='%*2int16%int16%int16':
#      %*2int16:     The * indicates a "discarded" column. This effectively
#                    skips the 16-byte header (assuming the header can be
#                    interpreted as two int16 units, or you can adjust this
#                    to match the exact byte size of your header).
#      %int16%int16: These read your IQ data (I and Q values) for each sample.
